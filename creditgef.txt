discordButton.Size = UDim2.new(0.2, 0, 0.05, 0) -- Mengurangi ukuran tombol
discordButton.Font = Enum.Font.SourceSansBold -- Mengatur font menjadi bold

-- Fungsi untuk menyalin teks ke clipboard
local function copyToClipboard(text)
    if game:GetService("GuiService"):GetWindowsPlatform() then
        game:GetService("GuiService"):CopyToClipboard(text)
    else
        -- Tambahkan kode untuk menyalin teks ke clipboard untuk platform non-Windows di sini
    end
end

-- Fungsi untuk menyalin teks Discord saat tombol diklik
local function copyDiscordLink()
    local discordLink = "https://discord.com/invite/5sG4jJwQ"
    copyToClipboard(discordLink)
end

-- Menjalankan fungsi copyDiscordLink saat tombol diklik
discordButton.MouseButton1Click:Connect(copyDiscordLink)
